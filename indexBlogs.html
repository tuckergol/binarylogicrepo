--- layout: default permalink: /sudoku title: 4x4 Binary Game --- %%js function generateBoard() { const size = 4; const board = Array.from({ length: size }, () => Array(size).fill(0)); // Fill the main diagonal with random values (0 or 1) for (let i = 0; i < size; i++) { board[i][i] = Math.floor(Math.random() * 2); } // Shuffle rows and columns to create a solvable puzzle for (let i = 0; i < size; i++) { const randomRow = Math.floor(Math.random() * size); const randomCol = Math.floor(Math.random() * size); // Swap rows const tempRow = [...board[i]]; board[i] = [...board[randomRow]]; board[randomRow] = tempRow; // Swap columns for (let j = 0; j < size; j++) { const temp = board[j][i]; board[j][i] = board[j][randomCol]; board[j][randomCol] = temp; } } return board; }
